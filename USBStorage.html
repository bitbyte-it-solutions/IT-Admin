# PowerShell script to block or unblock USB storage devices via Local Group Policy (Registry edits) function Block-USB { Write-Host "Blocking USB storage devices..." -ForegroundColor Red # Create the necessary registry path if it doesn't exist $regPath = "HKLM:\SOFTWARE\Policies\Microsoft\Windows\RemovableStorageDevices" if (-not (Test-Path $regPath)) { New-Item -Path $regPath -Force | Out-Null } # Set Deny All for USB storage devices (Device Class GUID: {53f56307-b6bf-11d0-94f2-00a0c91efb8b}) New-Item -Path "$regPath\{53f56307-b6bf-11d0-94f2-00a0c91efb8b}" -Force | Out-Null Set-ItemProperty -Path "$regPath\{53f56307-b6bf-11d0-94f2-00a0c91efb8b}" -Name "Deny_All" -Value 1 Write-Host "USB storage devices have been blocked." -ForegroundColor Green } function Unblock-USB { Write-Host "Unblocking USB storage devices..." -ForegroundColor Yellow $regPath = "HKLM:\SOFTWARE\Policies\Microsoft\Windows\RemovableStorageDevices\{53f56307-b6bf-11d0-94f2-00a0c91efb8b}" if (Test-Path $regPath) { Remove-Item -Path $regPath -Recurse -Force Write-Host "USB storage devices have been unblocked." -ForegroundColor Green } else { Write-Host "USB policy was not set. Nothing to unblock." -ForegroundColor Gray } } # Ensure the script is run as Administrator if (-not ([Security.Principal.WindowsPrincipal][Security.Principal.WindowsIdentity]::GetCurrent()).IsInRole(` [Security.Principal.WindowsBuiltInRole] "Administrator")) { Write-Host "Please run this script as Administrator." -ForegroundColor Red exit } # Display menu Write-Host "`nSelect an option:" Write-Host "1. Block USB storage devices" Write-Host "2. Unblock USB storage devices" $choice = Read-Host "Enter your choice (1 or 2)" switch ($choice) { "1" { Block-USB } "2" { Unblock-USB } default { Write-Host "Invalid selection. Exiting..." -ForegroundColor Red } }
